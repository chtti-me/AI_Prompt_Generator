建立 src/services/ai_service.py - AI 服務的核心！

📄 src/services/ai_service.py (簡化版)

"""AI 服務統一接口"""

import logging
import openai
import google.generativeai as genai
from config import config

logger = logging.getLogger(__name__)


class AIService:
    """AI 服務管理器"""
    
    def __init__(self):
        # 初始化 OpenAI
        if config.OPENAI_API_KEY:
            openai.api_key = config.OPENAI_API_KEY
            self.openai_client = openai
            logger.info("OpenAI 服務已初始化")
        
        # 初始化 Gemini
        if config.GOOGLE_AI_API_KEY:
            genai.configure(api_key=config.GOOGLE_AI_API_KEY)
            self.gemini_model = genai.GenerativeModel('gemini-2.0-flash-exp')
            logger.info("Gemini 服務已初始化")
    
    def generate_prompt(self, requirements, provider=None):
        """生成提示詞"""
        prompt = self._build_generation_prompt(requirements)
        
        if provider == 'gemini' and self.gemini_model:
            response = self.gemini_model.generate_content(prompt)
            return response.text
        else:
            return "請在 .env 配置 AI API Key"
    
    def _build_generation_prompt(self, requirements):
        """建構生成提示詞的 prompt"""
        return f"""請生成一個提示詞：
專案類型: {requirements.get('project_type', '未指定')}
主要功能: {', '.join(requirements.get('features', []))}
詳細程度: {requirements.get('detail_level', '中等')}
"""
    
    def is_available(self, provider=None):
        """檢查 AI 服務是否可用"""
        if provider == 'gemini':
            return hasattr(self, 'gemini_model')
        return False

程式碼說明：

  • ✅ __init__ - 初始化時自動連接可用的 AI 服務
  • ✅ generate_prompt - 根據需求生成提示詞
  • ✅ _build_generation_prompt - 建構發送給 AI 的 prompt
  • ✅ is_available - 檢查服務是否可用

"""提示詞模型"""

from sqlalchemy import Column, Integer, String, Text, DateTime
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime

Base = declarative_base()


class Prompt(Base):
    __tablename__ = 'prompts'
    
    id = Column(Integer, primary_key=True)
    title = Column(String(200), nullable=False)
    content = Column(Text, nullable=False)
    project_type = Column(String(50))
    created_at = Column(DateTime, default=datetime.now)
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now)
    
    def to_dict(self):
        return {
            'id': self.id,
            'title': self.title,
            'content': self.content,
            'project_type': self.project_type,
            'created_at': self.created_at.isoformat()
        }

📄 src/utils/db_manager.py

"""資料庫管理工具"""

from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
import logging

logger = logging.getLogger(__name__)


class DatabaseManager:
    def __init__(self, database_url):
        self.engine = create_engine(database_url)
        self.Session = sessionmaker(bind=self.engine)
        logger.info(f"資料庫已連接")
    
    def create_tables(self):
        from src.models.prompt import Base
        Base.metadata.create_all(self.engine)
        logger.info("資料表建立完成")


def init_db(database_url):
    db = DatabaseManager(database_url)
    db.create_tables()
    return db
